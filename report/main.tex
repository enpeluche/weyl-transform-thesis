\begin{proposition}

    \begin{itemize}

\begin{preuve}

\textbf{Preuve de $S_g S_h = S_{g \circ h}$.} \vspace{5pt}

$$S_gS_hf(x) = S_g(S_hf)(x) = \int_{\R^n} g(x,y) S_hf(y)dy$$
$$ = \int_{\R^n} g(x,y) \left(\int_{\R^n} h(y,z) f(z) dz\right) dy = \int_{\R^n} \int_{\R^n} g(x,y) h(y,z) f(z) dz dy$$\vspace{5pt}
